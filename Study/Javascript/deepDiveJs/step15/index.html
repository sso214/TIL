# 15장. let, const 키워드와 블록 레벨 스코프

# 1. var 키워드로 선언한 변수의 문제점

# 1-1. 변수 중복 선언 허용

var 키워드로 선언한 변수를 중복 선언 시 초기화문 유무에 따라 다르게 동작함.

  • 초기화문이 있는 변수 선언문 : 자바스크립트 엔진에 의해 var 키워드가 없는 것처럼 동작
  • 초기화문이 없는 변수 선언문 : 무시됨. 에러는 발생하지 않음

이 때문에 의도치 않게 먼저 선언된 변수 값이 변경되는 부작용이 발생함

var x = 1;
var y = 1;

//var 키워드로 선언된 변수는 같은 스코프 내 중뽁 선언을 허용함
var x = 100; //초기화문이 있는 변수 선언문은 var 키워드가 없는 것처럼 동작함
var y; //초기화문이 없는 변수 선언문은 무시함

console.log(x); //100
console.log(y); //1

# 1-2. 함수 레벨 스코프

var 키워드로 선언한 변수는 오로지 함수의 코드 블록만을 지역 스코프로 인정함
따라 함수 외부에서 var 키워드로 선언한 변수는 모두 전역 변수가 됨

함수 레벨 스코프는 전역 변수를 남발할 가능성 높임.
때문에 의도치 않게 전역 변수가 중복 선언되는 경우 발생함

var x = 1;

if(true) {
    //x는 전역 변수. 이미 선언된 전역 변수 x가 있으므로 중복 선언 -> 의도치 않게 변수값 변경됨
    var x = 10;
}
console.log(x); //10

# 1-3. 변수 호이스팅

var 키워드로 변수 선언 시 변수 호이스팅에 의해 변수 선언문이 스코프의 선두로 끌어 올려진 것처럼 동작함
(즉, 변수 선언문 이전에 참조 가능. 할당문 이전에 변수를 참조하면 언제나 undefined 반환함)

변수 선언문 이전에 변수를 참조하는 것은 함수 호이스팅으로 인해 에러를 발생시키지 않지만
프로그램의 흐름상 맞지 않고 가독성을 떨어뜨리고 오류 발생 여지가 있음

//이 시점에는 이미 젼수 호이스팅에 의해 foo 변수가 선언됨 (1. 선언 단계)
//변수 foo는 undefined로 초기화됨 (2. 초기화 단계)
console.log(foo); //undefined

//변수에 값을 할당 (3. 할당 단계)
foo = 123;

console.log(foo); //123

//변수 선언은 런타임 이전에 자바스크립트 엔진에 의해 암묵적으로 실행됨
var foo;

# 2. let 키워드

var 키워드의 단점을 보완하기 위해 ES6에서 새로운 변수 선언 키워드인 let, const가 도입됨

# 2-1. 변수 중복 선언 금지

let 키워드로 이름이 같은 변수 중복 선언시 문법 에러가 발생함

var foo = 123;
//var 키워드로 선언된 변수는 같은 스코프 내 중복 선언 허용
var foo = 456; //자바스크립트 엔진에 의해 var 키워드 없는 것처럼 동작

let bar = 123;
//let이나 const로 선언된 변수는 같은 스코프 내에서 중복 선언 허용하지 않음
let bar = 456; //SyntaxError

# 2-2. 블록 레벨 스코프

let 키워드로 선언한 변수는 모든 코드 블록 (함수, if문, for문, while문, try/catch문 등)을
지역 스코프로 인정하는 블록 레벨 스코프를 따름

함수도 코드 블록이므로 스코프를 만듬.
이때 함수 내의 코드 블록은 함수 레벨 스코프에 중첩됨

let foo = 1; //전역 변수

{
    let foo = 2; //지역 변수
    let bar = 3; //지역 변수
}

console.log(foo); //1
console.log(bar); //ReferenceError

# 2-3. 변수 호이스팅

let 키워드로 선언한 변수는 함수 호이스팅이 발생하지 않는 것처럼 동작해서
변수 선언문 이전에 참조 시 참조 에러가 발생함

console.log(foo); //ReferenceError
let foo;

# <var 키워드로 변수 선언>

런타임 이전에 자바스크립트 엔진에 의해 암묵적으로 선언과 초기화가 한번에 진행됨

  1. 선언 단계에서 스코프(실행 컨텍스트의 렉시컬 환경)에 변수 식별자를 등록해
    자바스크립트 엔진에 변수의 존재를 알림
  2. 즉시 초기화 단계에서 undefined로 변수를 초기화함
    따라 선언문 이전에 변수에 접근해도 스코프에 변수가 존재하기 때문에 에러가 발생하지 않음
    (다만 undefined를 반환)
  3. 이후 변수 할당문에 도달하면 값이 할당됨
console.log(foo); //undefined

var foo;
console.log(foo); //undefined

foo = 1;
console.log(foo); //1

# <let 키워드로 변수 선언>

let 키워드로 선언한 변수는 선언 단계와 초기화 단계가 분리되어 진행됨
즉, 런타임 이전에 자바스크립트 엔진에 의해 암묵적으로 선언 단계가 먼저 실행되고
초기화 단계는 변수 선언문에 도달했을 때 실행됨
따라 초기화 단계가 실행되기 전 변수에 접근하려고 하면 참조 에러 발생함

let 키워드로 선언한 변수는 스코프의 시작 지점부터 초기화 단계 시작 지점(변수 선언문)까지 변수 참조 불가
일시적 사각지대(TDZ) : 스코프의 시작 지점부터 초기화 시작 지점까지 변수를 참조할 수 없는 구간

//런타임 이전에 선언 단계 실행됨. 아직 변수가 초기화되지 않음
//초기화 이전의 일시적 사각지대에서는 변수 참조 불가
console.log(foo); //ReferenceError

let foo; //변수 선언문에서 초기화 단계 실행됨
console.log(foo); //undefined

foo = 1; //할당문에서 할당 단계 실행됨
console.log(foo); //1

따라 let 키워드로 선언한 변수는 변수 호이스팅이 발생하지 않는 것처럼 보이지만 그렇지 않음
만약 변수 호이스팅이 발생하지 않는다면 아래 코드는 전역 변수 foo의 값을 출력해야 함
하지만 let 키워드로 선언한 변수도 여전히 호이스팅이 발생하기 때문에 참조 에러가 발생함

let foo = 1; //전역 변수
{
    console.log(foo); //ReferenceError
    let foo = 2; //지역 변수
}

자바스크립트는 ES6에서 도입된 let, cosnt를 포함해서
모든 선언 (var, let, const, function, function*, class 등)을 호이스팅 함
단, ES6에서 도입된 let, const, class를 사용한 선언문은 호이스팅이 발생하지 않는 것처럼 동작함

# 2-4. 전역 객체와 let

var 키워드로 선언한 전역 변수와 전역 함수, 암묵적 전역(선언하지 않은 변수에 값을 할당)은
전역 객체 window의 프로퍼티가 됨
(전역 객체의 프로퍼티를 참조 시 window를 생략 가능)

let 키워드로 선언한 전역 변수는 전역 객체의 프로퍼티가 아님.
let 전역 변수는 보이지 않는 개념적인 블록 (전역 렉시컬 환경의 선언적 환경 레코드) 내에 존재하게 됨

var x = 1; //전역 변수
y = 2; //암묵적 전역
function foo(){} //전역 함수

console.log(window.x); //1
console.log(x); //1
console.log(window.y); //2
console.log(y); //2
console.log(window.foo); //f foo(){}
console.log(foo); //f foo(){}

let z = 1;
console.log(window.z); //undefined
console.log(z); //1

# 3. const 키워드

const 키워드는 상수를 선언하기 위해 사용함. 하지만 반드시 상수만을 위해 사용하지는 않음
const 키워드의 특징은 대부분 let 키워드와 동일하므로 let 키워드와 다른점을 중심으로 살펴봄

# 3-1. 선언과 초기화

  • const 키워드로 선언한 변수는 반드시 선언과 동시에 초기화해야 함. 그렇지 않으면 문법 에러 발생
  • const 키워드로 선언한 변수는 let 변수처럼 블록 레벨 스코프를 가지며 변수 호이스팅이 발생하지 않는 것처럼 동작
const foo = 1; 
const bar; //SyntaxError

//변수 호이스팅이 발생하지 않는 것처럼 동작
{
   console.log(test1); //ReferenceError
   const test1 = 1;
   console.log(test1); //1
}

//볼록 레벨 스코프를 가짐
console.log(test1); //ReferenceError

# 3-2. 재할당 금지

var과 let 키워드로 선언한 변수는 재할당이 가능하지만
const 키워드로 선언한 변수는 재할당이 금지됨

const foo = 1;
foo = 2; //TypeError

# 3-3. 상수

const 키워드로 선언한 변수에 원시 값을 할당한 경우 변수 값을 변경할 수 없음
(원시 값은 재할당밖에 값을 변경할 방법이 없기 때문)
이런 특징을 이용해 const 키워드를 상수를 표현하는데 사용하기도 함

상수 : 재할당이 금지된 변수. 변수의 상대 개념
상수도 값을 저장하기 위한 메모리 공간이 필요하므로 변수라고 할 수 있음.
단, 변수는 언제든 재할당을 통해 변수 값 변경이 가능하지만 상수는 재할당이 금지됨

상수는 상태 유지와 가독성, 유지보수의 편의를 위해 적극적으로 사용하는 것을 권장
일반적으로 상수명은 대문자 스네이크 케이스로 선언해 상수임을 명확히 나타냄

let preTaxPrice = 100;

//0.1이 어떤 의미로 사용되었는지 의미 명확하게 알기 어려워 가독성 떨어짐
let afterTaxPrice = preTaxPrice + (preTaxPrice * 0.1);
console.log(afterTaxPrice); //110

const TAX_RATE = 0.1;
//0.1은 고정으로 사용하는 값이기 떄문에 상수로 정의하면 값의 의미 쉽게 파악 가능함 
//또한 하나의 상수를 여러곳에서 쓸 수 있기 떄문에 유지보수성 향상됨
let afterTaxPrice = preTaxPrice + (preTaxPrice * TAX_RATE);
console.log(afterTaxPrice); //110

# 3-4. const 키워드와 객체

const 키워드로 선언된 변수에 원시 값을 할당한 경우 값 변경이 불가함.
하지만 const 키워드로 선언된 변수에 객체를 할당한 경우 값 변경이 가능함.
변경 가능한 값인 객체는 재할당 없이도 직접 변경이 가능하기 때문

const 키워드는 재할당을 금지할 뿐 불변을 의미하지는 않음
따라 새로운 값을 재할당하는 것은 불가능하지만
프로퍼티 동적 생성, 삭제, 프로퍼티 값의 변경을 통해 객체를 변경하는 것은 가능함.
(이때 객체가 변경되더라도 변수에 할당된 참조값은 변경되지 않음)

const person = {
    name: 'Leo'
};
person.name = 'Park';
console.log(person); //{name: 'Park'}

# 4. var vs. let vs. const

변수 선언에는 기본적으로 const를 사용하고 let은 재할당이 필요한 경우에 한정해 사용하는 것이 좋음.
const 키워드를 사용하면 의도치 않은 재할당을 방지할 수 있기 때문에 더 안전함

변수 선언 시점에는 재할당이 필요한지 잘 모르는 경우가 많음.
그리고 객체는 의외로 재할당하는 경우가 드물기 때문에 변수 선언할 때는 일단 const 키워드를 사용.
후에 재할당이 필요한 경우로 판단되면 그때 let 키워드로 변경해도 됨

  • ES6를 사용한다면 var 키워드는 사용 X
  • 재할당이 필요한 경우에 한정해 let 키워드를 사용. 이때 변수의 스코프는 최대한 좁게 만듬
  • 변경이 발생하지 않고 읽기 전용으로 사용하는 원시 값과 객체에는 const 키워드 사용.
    const 키워드는 재할당을 금지하므로 var, let 키워드보다 안전함
